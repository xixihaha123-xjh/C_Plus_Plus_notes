struct node
{
    int val;
    struct node* next;
};
typedef struct node MyListNode;

class MyLinkedList
{
private:
    MyListNode* m_head;               // 维护一个头节点
    MyListNode* m_tail;               // 维护一个尾节点
    int m_size;                       // 链表长度
public:
    /** Initialize your data structure here. */
    MyLinkedList()
    {
        m_head = new MyListNode;
        m_tail = m_head;              // 头节点也是尾结点，头节点不存储数据
        m_size = 0;                   // 头节点不计算在链表长度之内
    }
    
    /** Get the value of the index-th node in the linked list. If the index is invalid, return -1. */
    int get(int index)
    {
        if(index < 0) return -1;
        MyListNode* curr = m_head;
        for(int i = 0; i <= index; i++)
        {
            curr = curr->next;
        }
        return curr->val;
    }
    
    /** Add a node of value val before the first element of the linked list. After the insertion, the new node will be the first node of the linked list. */
    void addAtHead(int val)
    {
        MyListNode* new_node = new MyListNode;
        
        new_node->next = m_head->next;
        m_head->next = new_node;
        new_node->val = val;
        
        m_size++;
        if(1 == m_size)
        {
            m_tail = new_node;
        }
    }
    
    /** Append a node of value val to the last element of the linked list. */
    void addAtTail(int val)
    {
        MyListNode* new_node = new MyListNode;
        m_tail->next =  new_node;
        new_node->val = val;
        m_size++;
        m_tail = new_node;
    }
    
    /** Add a node of value val before the index-th node in the linked list. If index equals to the length of linked list, the node will be appended to the end of linked list. If index is greater than the length, the node will not be inserted. */
    void addAtIndex(int index, int val)
    {
        if(index > m_size) return;
        if(index < 0)
        {
            addAtHead(val);
            return;
        }

        if(index == m_size)
        {
            addAtTail(val);
            return;
        }

        MyListNode* new_node = new MyListNode;
        MyListNode* curr = m_head;
        for(int i = 0; i < index; i++)
        {
            curr = curr->next;
        }
        new_node->next = curr->next;
        curr->next = new_node;
        new_node->val = val;
        m_size++;
    }
    
    /** Delete the index-th node in the linked list, if the index is valid. */
    void deleteAtIndex(int index)
	{
		if (index < 0 || index > m_size - 1) return;

		MyListNode* prev = m_head;
		MyListNode* curr = m_head->next;
		
		for (int i = 0; i < index; i++)
		{
			prev = curr;
			curr = curr->next;
		}

		prev->next = curr->next;
		delete(curr);
        curr == nullptr;
		m_size--;
	}
};
